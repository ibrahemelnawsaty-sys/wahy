📝 مثال: API Controller 

<?php

namespace App\Http\Controllers\Api\V1;

use App\Http\Controllers\Controller;
use App\Http\Requests\StoreActivityRequest;
use App\Http\Resources\ActivityResource;
use App\Services\ActivityService;
use Illuminate\Http\JsonResponse;
use Illuminate\Http\Resources\Json\AnonymousResourceCollection;

class ActivityController extends Controller
{
    public function __construct(
        private ActivityService $activityService
    ) {}

    /**
     * Display a listing of activities.
     *
     * @return AnonymousResourceCollection
     */
    public function index(): AnonymousResourceCollection
    {
        $activities = $this->activityService->getAllActivities(
            perPage: request('per_page', 15)
        );

        return ActivityResource::collection($activities);
    }

    /**
     * Store a newly created activity.
     *
     * @param StoreActivityRequest $request
     * @return JsonResponse
     */
    public function store(StoreActivityRequest $request): JsonResponse
    {
        $activity = $this->activityService->createActivity(
            $request->validated()
        );

        return response()->json([
            'message' => 'Activity created successfully',
            'data' => new ActivityResource($activity)
        ], 201);
    }

    /**
     * Display the specified activity.
     *
     * @param int $id
     * @return JsonResponse
     */
    public function show(int $id): JsonResponse
    {
        $activity = $this->activityService->getActivityById($id);

        return response()->json([
            'data' => new ActivityResource($activity)
        ]);
    }

    /**
     * Update the specified activity.
     *
     * @param StoreActivityRequest $request
     * @param int $id
     * @return JsonResponse
     */
    public function update(StoreActivityRequest $request, int $id): JsonResponse
    {
        $activity = $this->activityService->updateActivity(
            $id,
            $request->validated()
        );

        return response()->json([
            'message' => 'Activity updated successfully',
            'data' => new ActivityResource($activity)
        ]);
    }

    /**
     * Remove the specified activity.
     *
     * @param int $id
     * @return JsonResponse
     */
    public function destroy(int $id): JsonResponse
    {
        $this->activityService->deleteActivity($id);

        return response()->json([
            'message' => 'Activity deleted successfully'
        ], 204);
    }
}

⚙️ مثال: Service Layer 

<?php

namespace App\Services;

use App\Models\Activity;
use App\Models\Submission;
use App\Models\User;
use App\Repositories\Contracts\ActivityRepositoryInterface;
use App\Events\ActivitySubmitted;
use Illuminate\Pagination\LengthAwarePaginator;
use Illuminate\Support\Facades\DB;

class ActivityService
{
    public function __construct(
        private ActivityRepositoryInterface $activityRepository,
        private BadgeService $badgeService
    ) {}

    /**
     * Get all activities with pagination.
     */
    public function getAllActivities(int $perPage = 15): LengthAwarePaginator
    {
        return $this->activityRepository->paginate($perPage);
    }

    /**
     * Create a new activity.
     */
    public function createActivity(array $data): Activity
    {
        return DB::transaction(function () use ($data) {
            $activity = $this->activityRepository->create($data);

            // Additional business logic here
            // e.g., create related records, send notifications, etc.

            return $activity;
        });
    }

    /**
     * Submit an activity.
     */
    public function submitActivity(int $activityId, User $student, array $answers): Submission
    {
        return DB::transaction(function () use ($activityId, $student, $answers) {
            // Create submission
            $submission = Submission::create([
                'activity_id' => $activityId,
                'user_id' => $student->id,
                'answers' => $answers,
                'status' => 'pending',
                'submitted_at' => now()
            ]);

            // Fire event for notifications
            event(new ActivitySubmitted($submission));

            // Check if student earned any badges
            $this->badgeService->checkAndAwardBadges($student);

            return $submission;
        });
    }

    /**
     * Get activity by ID.
     */
    public function getActivityById(int $id): Activity
    {
        return $this->activityRepository->findOrFail($id);
    }

    /**
     * Update activity.
     */
    public function updateActivity(int $id, array $data): Activity
    {
        $activity = $this->getActivityById($id);

        return $this->activityRepository->update($activity, $data);
    }

    /**
     * Delete activity.
     */
    public function deleteActivity(int $id): bool
    {
        $activity = $this->getActivityById($id);

        return $this->activityRepository->delete($activity);
    }
}

💾 مثال: Repository Pattern

Repository Pattern يفصل Business Logic عن Data Access Layer

📌 Interface (Contract)

<?php

namespace App\Repositories\Contracts;

use App\Models\Activity;
use Illuminate\Pagination\LengthAwarePaginator;

interface ActivityRepositoryInterface
{
    public function all(): Collection;
    public function paginate(int $perPage = 15): LengthAwarePaginator;
    public function find(int $id): ?Activity;
    public function findOrFail(int $id): Activity;
    public function create(array $data): Activity;
    public function update(Activity $activity, array $data): Activity;
    public function delete(Activity $activity): bool;
    public function findByLesson(int $lessonId): Collection;
}

📌 Implementation

<?php

namespace App\Repositories\Eloquent;

use App\Models\Activity;
use App\Repositories\Contracts\ActivityRepositoryInterface;
use Illuminate\Database\Eloquent\Collection;
use Illuminate\Pagination\LengthAwarePaginator;

class ActivityRepository implements ActivityRepositoryInterface
{
    public function all(): Collection
    {
        return Activity::with(['lesson', 'meaning'])->get();
    }

    public function paginate(int $perPage = 15): LengthAwarePaginator
    {
        return Activity::with(['lesson', 'meaning'])
            ->latest()
            ->paginate($perPage);
    }

    public function find(int $id): ?Activity
    {
        return Activity::with(['lesson', 'meaning', 'submissions'])
            ->find($id);
    }

    public function findOrFail(int $id): Activity
    {
        return Activity::with(['lesson', 'meaning', 'submissions'])
            ->findOrFail($id);
    }

    public function create(array $data): Activity
    {
        return Activity::create($data);
    }

    public function update(Activity $activity, array $data): Activity
    {
        $activity->update($data);
        return $activity->fresh();
    }

    public function delete(Activity $activity): bool
    {
        return $activity->delete();
    }

    public function findByLesson(int $lessonId): Collection
    {
        return Activity::where('lesson_id', $lessonId)->get();
    }
}

⭐ Best Practices

1️⃣ PSR-12 Code Standard

  • استخدام 4 spaces للـ indentation
  • Opening braces على نفس السطر للـ methods
  • استخدام type hints لجميع parameters و return types
  • Camel case للـ methods، Pascal case للـ classes

2️⃣ SOLID Principles

  • Single Responsibility: كل class لها مسؤولية واحدة
  • Open/Closed: مفتوحة للامتداد، مغلقة للتعديل
  • Liskov Substitution: استخدام Interfaces بدلاً من Concrete Classes
  • Interface Segregation: Interfaces صغيرة ومحددة
  • Dependency Inversion: Depend on abstractions

3️⃣ PHPStan Level 8

  • أعلى مستوى من الـ static analysis
  • Type safety في جميع الأكواد
  • No mixed types إلا عند الضرورة
  • PHPDoc blocks لجميع methods

4️⃣ Performance Optimization

  • Eager Loading مع with() لتجنب N+1 queries
  • Database Indexing على الحقول المستخدمة في WHERE/JOIN
  • Redis Cache للبيانات المستخدمة بكثرة
  • Queue للعمليات الطويلة (emails, notifications)
