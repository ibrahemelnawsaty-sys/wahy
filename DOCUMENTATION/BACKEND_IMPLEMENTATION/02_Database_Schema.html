users

الحقل النوع الوصف قيود
id BIGINT UNSIGNED المفتاح الأساسي PRIMARY KEY, AUTO_INCREMENT
name VARCHAR(255) اسم المستخدم الكامل NOT NULL
email VARCHAR(255) البريد الإلكتروني UNIQUE, NOT NULL, INDEX
password VARCHAR(255) كلمة المرور المشفرة NOT NULL
role_id BIGINT UNSIGNED الدور (Role) FOREIGN KEY → roles(id)
school_id BIGINT UNSIGNED المدرسة التابع لها FOREIGN KEY → schools(id), NULLABLE
branch_id BIGINT UNSIGNED الفرع التابع له FOREIGN KEY → branches(id), NULLABLE
classroom_id BIGINT UNSIGNED الفصل الدراسي FOREIGN KEY → classrooms(id), NULLABLE
points INT UNSIGNED النقاط المكتسبة DEFAULT 0
coins INT UNSIGNED العملات الافتراضية DEFAULT 0
status ENUM active, pending, suspended DEFAULT 'pending'
email_verified_at TIMESTAMP وقت تأكيد البريد NULLABLE
created_at TIMESTAMP وقت الإنشاء DEFAULT CURRENT_TIMESTAMP
updated_at TIMESTAMP وقت آخر تحديث ON UPDATE CURRENT_TIMESTAMP 
-- Migration
Schema::create('users', function (Blueprint $table) {
    $table->id();
    $table->string('name');
    $table->string('email')->unique();
    $table->string('password');
    $table->foreignId('role_id')->constrained()->onDelete('restrict');
    $table->foreignId('school_id')->nullable()->constrained()->onDelete('set null');
    $table->foreignId('branch_id')->nullable()->constrained()->onDelete('set null');
    $table->foreignId('classroom_id')->nullable()->constrained()->onDelete('set null');
    $table->unsignedInteger('points')->default(0);
    $table->unsignedInteger('coins')->default(0);
    $table->enum('status', ['active', 'pending', 'suspended'])->default('pending');
    $table->timestamp('email_verified_at')->nullable();
    $table->timestamps();
    
    $table->index(['email', 'status']);
    $table->index('role_id');
    $table->index('school_id');
});

📚

values → concepts → meanings → lessons

هيكل المحتوى التعليمي الهرمي

1. values (القيم الأساسية)
  • id, title, description, icon, order, status, timestamps
2. concepts (المفاهيم)
  • id, value_id (FK), title, description, order, timestamps
3. meanings (المعاني)
  • id, concept_id (FK), title, description, order, timestamps
4. lessons (الدروس)
  • id, meaning_id (FK), title, content (TEXT), video_url, duration, order, timestamps

🔗Entity Relationship Diagram

العلاقات الرئيسية

users ↔ roles (Many-to-One)

كل مستخدم له دور واحد (Super Admin, School Admin, Teacher, Student, Parent)

users ↔ schools (Many-to-One)

المستخدمون ينتمون لمدرسة واحدة (ما عدا Super Admin)

schools → branches → classrooms (Hierarchical)

هيكل هرمي: مدرسة → فروع متعددة → فصول متعددة

values → concepts → meanings → lessons (Hierarchical)

هيكل المحتوى التعليمي الهرمي المكون من 4 مستويات

lessons → activities (One-to-Many)

كل درس يحتوي على أنشطة متعددة

activities → submissions (One-to-Many)

كل نشاط له حلول متعددة من الطلاب

users → submissions (One-to-Many)

كل طالب يقدم حلول متعددة

users ↔ badges (Many-to-Many)

الطلاب يحصلون على شارات متعددة، والشارة تُمنح لطلاب متعددين
