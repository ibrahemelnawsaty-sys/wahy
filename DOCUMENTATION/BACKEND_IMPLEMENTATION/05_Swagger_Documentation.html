⚙️Installation & Setup

1. Install L5-Swagger

composer require darkaonline/l5-swagger

php artisan vendor:publish --provider "L5Swagger\L5SwaggerServiceProvider"

2. Configure config/l5-swagger.php

return [
    'api' => [
        'title' => 'منصة قيمّ API Documentation',
    ],
    'routes' => [
        'api' => 'api/documentation',
    ],
    'paths' => [
        'docs' => storage_path('api-docs'),
        'docs_json' => 'api-docs.json',
        'annotations' => [
            base_path('app'),
        ],
    ],
];

3. Generate Documentation

php artisan l5-swagger:generate

use OpenApi\Annotations as OA;

 * @OA\Server(
 *     url="https://api.qiyamm.com/api/v1",
 *     description="Production Server"
 * )

 *     type="http",
 *     scheme="bearer",
 *     bearerFormat="JWT"
 * )
 */
class ActivityController extends Controller
{
    /**
     * @OA\Get(
     *     path="/activities",
     *     summary="Get all activities",
     *     tags={"Activities"},
     *     security={{"bearerAuth":{}}},

     *         name="page",
     *         in="query",
     *         description="Page number",
     *         required=false,
     *         @OA\Schema(type="integer", default=1)
     *     ),
     *     @OA\Parameter(
     *         name="per_page",
     *         in="query",
     *         description="Items per page",
     *         required=false,
     *         @OA\Schema(type="integer", default=15)
     *     ),
     *     @OA\Parameter(
     *         name="lesson_id",
     *         in="query",
     *         description="Filter by lesson ID",
     *         required=false,
     *         @OA\Schema(type="integer")
     *     ),
     *     @OA\Response(
     *         response=200,
     *         description="Successful operation",
     *         @OA\JsonContent(
     *             type="object",
     *             @OA\Property(property="data", type="array",
     *                 @OA\Items(
     *                     @OA\Property(property="id", type="integer", example=1),
     *                     @OA\Property(property="title", type="string", example="نشاط الأمانة"),

     *                 @OA\Property(property="total", type="integer", example=100)
     *             )
     *         )
     *     ),
     *     @OA\Response(
     *         response=401,
     *         description="Unauthenticated"
     *     )
     * )
     */
    public function index()

        // Implementation...
    }

    /**
     * @OA\Post(
     *     path="/activities/{id}/submit",
     *     summary="Submit activity solution",
     *     tags={"Activities"},

     *         name="id",

     *                 @OA\Property(property="submission_id", type="integer", example=123),

     *             )
     *         )
