🔐 Authentication

POST تسجيل الدخول

// Request POST /api/v1/auth/login Content-Type: application/json { "email": "teacher@school.com", "password": "password123" } // Response { "success": true, "data": { "user": { "id": 123, "name": "أحمد محمد", "email": "teacher@school.com", "role": "teacher" }, "token": "eyJ0eXAiOiJKV1QiLCJhbGc...", "expires_in": 3600 } } // Usage in JavaScript const login = async (email, password) => { const response = await fetch('/api/v1/auth/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email, password }) }); const data = await response.json(); localStorage.setItem('token', data.data.token); return data; };

👨‍🎓 Students API

GET قائمة الطلاب

// Request GET /api/v1/students?class_id=5&page=1 Authorization: Bearer {token} // Response { "success": true, "data": [ { "id": 1, "name": "محمد أحمد", "class": "الصف الأول أ", "parent": "أحمد علي", "attendance_rate": 95.5 }, // ... more students ], "meta": { "current_page": 1, "total": 30, "per_page": 15 } } // Usage with Axios import axios from 'axios'; const getStudents = async (classId, page = 1) => { const response = await axios.get('/api/v1/students', { params: { class_id: classId, page }, headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}` } }); return response.data; };

📊 Attendance API

POST تسجيل الحضور

// Request POST /api/v1/attendance Authorization: Bearer {token} Content-Type: application/json { "class_id": 5, "date": "2026-01-20", "students": [ { "student_id": 1, "status": "present" }, { "student_id": 2, "status": "present" }, { "student_id": 3, "status": "absent", "reason": "مرض" } ] } // Response { "success": true, "message": "تم حفظ الحضور بنجاح", "data": { "total_students": 30, "present": 28, "absent": 2, "attendance_rate": 93.3 } } // Vue 3 Composable Example import { ref } from 'vue'; import { useApi } from '@/composables/useApi'; export function useAttendance() { const { post } = useApi(); const loading = ref(false); const saveAttendance = async (classId, date, students) => { loading.value = true; try { const response = await post('/attendance', { class_id: classId, date, students }); return response.data; } finally { loading.value = false; } }; return { saveAttendance, loading }; }

📝 Grades API

GET درجات الطالب

// Request GET /api/v1/students/123/grades?semester=1 Authorization: Bearer {token} // Response { "success": true, "data": { "student": { "id": 123, "name": "محمد أحمد" }, "subjects": [ { "name": "الرياضيات", "grades": [ { "type": "quiz", "score": 18, "max": 20 }, { "type": "midterm", "score": 45, "max": 50 }, { "type": "final", "score": 88, "max": 100 } ], "average": 89.4 }, // ... more subjects ], "overall_average": 87.5 } } // React Hook Example import { useState, useEffect } from 'react'; function useStudentGrades(studentId, semester) { const [grades, setGrades] = useState(null); const [loading, setLoading] = useState(true); useEffect(() => { fetch(`/api/v1/students/${studentId}/grades?semester=${semester}`, { headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}` } }) .then(res => res.json()) .then(data => { setGrades(data.data); setLoading(false); }); }, [studentId, semester]); return { grades, loading }; }

💡 أفضل الممارسات

  • استخدم HTTPS دائماً: لحماية البيانات المنقولة
  • احفظ Token بشكل آمن: استخدم httpOnly cookies أو localStorage بحذر
  • Rate Limiting: احترم حدود الطلبات (1000 req/hour)
  • Error Handling: تعامل مع جميع حالات الأخطاء
  • Pagination: استخدم الترقيم للبيانات الكبيرة
  • Caching: استخدم Cache للبيانات التي لا تتغير كثيراً
